using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication1
{
    class Program
    {
        static void Main(string[] args)
        {
            /*Uma competição de natação está dividida em 3 provas.
             * Nessa competição estão inscritos 5 nadadores , e todos os nadadores participam de todas as provas.
             * Faça um programa que leia os resultados de cada prova -- o tempo de cada nadador na prova -- e determine :
             * 
             * 1) O nadador campeão ( menor tempo total )
             * 2) O nadador vencedor de cada prova (menor tempo na prova)
             * 
             * O programa começa com a leitura dos nomes dos 5 participantes . Em seguida , você deve ler os tempos de cada prova 
             * (ou seja , os tempos de cada nadador da prova ) para todas as provas.O tempo é dado em segundos , com precisão de duas casas decimais.
             * 
             * Veja o exemplo de entrada no quadro :
             * 
             * 
             * */
            string [] nomes = new string [5];
            int[,] nadadoreprova = new int[5, 3];
            float tempo;
            string nome;
            float melhortempo = -16;
            int melhor=0;
            for (int i = 0; i < 5; i++)
            {
                Console.WriteLine("Digite os nomes");
                nome = Convert.ToString(Console.ReadLine());
                nomes[i] = nome;
            }

            for (int j = 0; j < 3; j++)
            {
                Console.WriteLine("Digite as notas da prova {0} :", j + 1);
                for (int y = 0; y < 5; y++)
                {
                    tempo = float.Parse(Console.ReadLine());
                    tempo = nadadoreprova[y, j];
                   
                }

            }
            for (int a = 0; a < 3; a++)
            {
                for (int x = 0; x < 5; x++)
                {
                     if(nadadoreprova[x,a] > melhortempo )
                    {
                        nadadoreprova[x,a] = melhor;
                        melhortempo = melhor;
                      
                   
                     }
                }
                Console.Write("{0} Ganhou a {1} prova" + "   ", nomes[a], a);
            }
            
        }
    }
}
